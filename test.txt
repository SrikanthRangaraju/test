import java.io.*;

public class ReplacePipeWithComma {

    public static void main(String[] args) {
        String inputFilePath = "path/to/input.csv";
        String outputFilePath = "path/to/output.csv";

        try (BufferedReader br = new BufferedReader(new FileReader(inputFilePath));
             BufferedWriter bw = new BufferedWriter(new FileWriter(outputFilePath))) {

            String line;
            while ((line = br.readLine()) != null) {
                // Replace all pipe symbols with commas
                String updatedLine = line.replace("|", ",");
                bw.write(updatedLine);
                bw.newLine();
            }

            System.out.println("Pipe symbols replaced with commas successfully!");

        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
