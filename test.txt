import java.io.*;
import java.util.*;

public class CommonSSNFinder {
    public static void main(String[] args) {
        // Paths to the input files
        String file1Path = "file1.txt";
        String file2Path = "file2.txt";
        String outputFilePath = "common_ids.txt";

        try {
            // Read SSN IDs from both files into sets
            Set<String> file1SSNs = readSSNsFromFile(file1Path);
            Set<String> file2SSNs = readSSNsFromFile(file2Path);

            // Find common SSN IDs
            file1SSNs.retainAll(file2SSNs);

            // Write the common SSNs to the output file
            writeSSNsToFile(outputFilePath, file1SSNs);

            System.out.println("Common SSN IDs have been written to: " + outputFilePath);
        } catch (IOException e) {
            System.err.println("Error: " + e.getMessage());
        }
    }

    private static Set<String> readSSNsFromFile(String filePath) throws IOException {
        Set<String> ssnSet = new HashSet<>();
        try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = reader.readLine()) != null) {
                ssnSet.add(line.trim());
            }
        }
        return ssnSet;
    }

    private static void writeSSNsToFile(String filePath, Set<String> ssns) throws IOException {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath))) {
            for (String ssn : ssns) {
                writer.write(ssn);
                writer.newLine();
            }
        }
    }
}
